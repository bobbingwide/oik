Array ( [script] => Array ( [default] => [values] => script-name [notes] => Handle for the jQuery script ) [selector] => Array ( [default] => [values] => selector [notes] => jQuery selector ) [method] => Array ( [default] => [values] => method [notes] => jQuery method to perform ) [debug] => Array ( [default] => false [values] => true [notes] => Use true when you want to debug the jQuery ) [windowload] => Array ( [default] => false [values] => true [notes] => Use true when the jQuery is to run when the window has loaded ) [parms] => Array ( [default] => [values] => parm=value1,parm2=value2 [notes] => Variable list of parameters ) [src] => Array ( [default] => [values] => ID|URL [notes] => ID or full URL of JavaScript ) [inline] => Array ( [default] => [values] => T|Y [notes] => Set to T=True or Y=Yes when the script tag must be inline ) )
